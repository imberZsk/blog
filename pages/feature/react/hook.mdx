## useLoading

```tsx
import { useState } from 'react'

export const useRequestLoading = () => {
  const [loading, setLoading] = useState(false)
  const executeRequest = async (promiseFn: VoidFunction) => {
    setLoading(true)
    const result = await promiseFn()
    setLoading(false)
  }
  return {
    loading,
    executeRequest
  }
}
```

#### useRef

useRef 让函数只执行一次

```jsx
function useConstructor(callback) {
  const init = useRef(true)
  if (init.current) {
    callback()
    init.current = false
  }
}
```
