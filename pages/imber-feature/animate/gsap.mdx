## gsap基础使用

官网`https://greensock.com/get-started/#tweening-basics`

动画曲线：`https://greensock.com/get-started/#easing`

```jsx
import gsap from 'gsap'
import { useEffect, useRef } from 'react'
import styles from './index.module.css'
const Home = () => {
  const boxRef = useRef(null)
  useEffect(() => {
    gsap.to(boxRef.current, {
      duration: 3,
      x: 100,
      height: '100px'
    })
  }, [])
  return <div ref={boxRef} className={styles.box}></div>
}

export default Home
```

```css
.box {
  height: 200px;
  width: 200px;
  background-color: pink;
}
```

## react-spring基础使用

```jsx
import { useSpring, animated } from '@react-spring/web'

export default function MyComponent() {
  const [springs, api] = useSpring(() => ({
    from: { x: 0 }
  }))

  const handleClick = () => {
    api.start({
      from: {
        x: 0
      },
      to: {
        x: 100
      }
    })
  }

  return (
    <animated.div
      onClick={handleClick}
      style={{
        width: 80,
        height: 80,
        background: '#ff6d6d',
        borderRadius: 8,
        ...springs
      }}
    />
  )
}
```
