// 处理倒计时函数
function countDown(startTime, endTime, perSecondCallback1, endCallback1, text) {
  var s = Math.floor((endTime - startTime) / 1000) - 1;
  var timer = setInterval(function () {
    if (s < 0) {
      endCallback1 && endCallback1();
      clearInterval(timer);
    } else {
      perSecondCallback1 && perSecondCallback1(s, text);
    }
    s--;
  }, 1000);
  return timer;
}
function perSecondCallback(leftTime, text) {
  var hours = Math.floor((leftTime % 86400) / 3600);
  var minutes =
    Math.floor((leftTime % 3600) / 60) < 10
      ? '0' + Math.floor((leftTime % 3600) / 60)
      : Math.floor((leftTime % 3600) / 60);
  var seconds = leftTime % 60 < 10 ? '0' + (leftTime % 60) : leftTime % 60;
  var target = hours + '时' + minutes + '分' + seconds + '秒';
  restTime = text + target;
}
